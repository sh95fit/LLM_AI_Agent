PART 0: 시작하기 전에

1. RAG(검색 증강 생성)란 무엇인가?

AI 챗봇(LLM)은 기본적으로 학습하지 않은 여러분 회사의 비공개 정보를 모릅니다. RAG(Retrieval-Augmented Generation)는 이를 해결하기 위해 마치 "오픈북 시험"을 보는 것과 같습니다.

  • 기존 챗봇: 머릿속에 있는 지식만으로 답변 (회사 정보 모름)
  • RAG 챗봇: 질문을 받으면 1) 회사 문서를 찾아보고(검색), 2) 찾은 내용을 참고해서 답변(생성)

PART 2: 외부 API 설정

1. Notion API 설정

  1. Notion My Integrations 접속 후 '새 API 통합 만들기' 클릭.
  2. 이름을 입력(예: RagBot)하고 제출.
  3. [내부 통합 시크릿] 키를 복사해둡니다. (이것이 NOTION_API_KEY)
  4. 연동하려는 Notion 페이지로 이동 → 우측 상단 ... 클릭 → 연결 추가 → 방금 만든 통합 앱 선택.
  5. 페이지 URL에서 데이터베이스 ID 추출: https://notion.so/myworkspace/{DATABASE_ID}?v=... (32자리 문자열)

2. Google OAuth 2.0 설정

  1. Google Cloud Console 접속 및 새 프로젝트 생성.
  2. 메뉴 → 'API 및 서비스' → 'OAuth 동의 화면' → '외부' 선택 후 만들기.
  3. 앱 이름, 이메일 등 필수 정보 입력 후 저장.
  4. '사용자 인증 정보' → '사용자 인증 정보 만들기' → 'OAuth 클라이언트 ID'.
  5. 애플리케이션 유형: 웹 애플리케이션.
  6. 승인된 리디렉션 URI: http://localhost:8000/auth/callback 추가.
  7. 생성된 클라이언트 ID와 클라이언트 보안 비밀(Secret) 복사.

# 동기 드라이버 URL로 변환 (Alembic은 내부적으로 동기 처리 필요 가능성 대비)
# 하지만 여기서는 AsyncEngine을 사용하여 비동기 마이그레이션 실행
config.set_main_option("sqlalchemy.url", DATABASE_URL)

def run_migrations_offline() -> None:
    url = config.get_main_option("sqlalchemy.url")
    context.configure(
        url=url,
        target_metadata=target_metadata,
        literal_binds=True,
        dialect_opts={"paramstyle": "named"},
    )
    with context.begin_transaction():
        context.run_migrations()

async def run_migrations_online() -> None:
    connectable = async_engine_from_config(
        config.get_section(config.config_ini_section),
        prefix="sqlalchemy.",
        poolclass=pool.NullPool,
    )

    async with connectable.connect() as connection:
        await connection.run_sync(do_run_migrations)

    await connectable.dispose()

def do_run_migrations(connection: Connection) -> None:
    context.configure(connection=connection, target_metadata=target_metadata)
    with context.begin_transaction():
        context.run_migrations()

if context.is_offline_mode():
    run_migrations_offline()
else:
    asyncio.run(run_migrations_online())

backend/core/vector_store.py
from qdrant_client import QdrantClient
from qdrant_client.http import models
from langchain_ollama import OllamaEmbeddings
from config import get_settings

settings = get_settings()

class VectorStore:
    def __init__(self):
        self.client = QdrantClient(host=settings.QDRANT_HOST, port=settings.QDRANT_PORT)
        self.embeddings = OllamaEmbeddings(
            base_url=settings.OLLAMA_BASE_URL,
            model=settings.OLLAMA_EMBED_MODEL
        )
        self.collection_name = settings.QDRANT_COLLECTION
        self._ensure_collection()

    def _ensure_collection(self):
        """컬렉션이 없으면 생성"""
        collections = self.client.get_collections().collections
        if not any(c.name == self.collection_name for c in collections):
            self.client.create_collection(
                collection_name=self.collection_name,
                vectors_config=models.VectorParams(size=768, distance=models.Distance.COSINE)
            )

    def add_texts(self, texts: list[str], metadatas: list[dict]):
        """텍스트 임베딩 후 저장"""
        embeddings = self.embeddings.embed_documents(texts)
        points = [
            models.PointStruct(
                id=idx,  # 실제 구현시 UUID 사용 권장
                vector=embedding,
                payload={"text": text, **metadata}
            )
            for idx, (text, embedding, metadata) in enumerate(zip(texts, embeddings, metadatas))
        ]
        # Qdrant에 업로드 (배치 처리는 생략, 간단 구현)
        # 실제로는 고유 ID 관리 필요. 여기서는 간단히 덮어쓰기 로직이 필요할 수 있음.
        # LangChain Qdrant wrapper를 쓰는 것이 편하지만, 구조 이해를 위해 직접 구현하거나 LangChain 연결
        pass 
        
    def get_langchain_store(self):
        """LangChain용 VectorStore 객체 반환"""
        from langchain_community.vectorstores import Qdrant
        return Qdrant(
            client=self.client,
            collection_name=self.collection_name,
            embeddings=self.embeddings
        )

vector_store = VectorStore()

    아래의 문맥(Context)을 사용하여 질문에 답변해주세요.
    답변은 한국어로 작성하고, 모르는 내용은 "문서에 내용이 없습니다"라고 하세요.
    
    
    {context}
    
    
    질문: {input}
    """)
    
    # 4. 체인 연결
    document_chain = create_stuff_documents_chain(llm, prompt)
    retrieval_chain = create_retrieval_chain(retriever, document_chain)
    
    return retrieval_chain

import redis
import json
from config import get_settings

settings = get_settings()

class ChatMemory:
    def __init__(self):
        self.redis = redis.Redis(
            host=settings.REDIS_HOST,
            port=settings.REDIS_PORT,
            db=settings.REDIS_DB,
            decode_responses=True
        )
        
    def add_message(self, session_id: str, role: str, content: str):
        """메시지 저장 (List 사용)"""
        message = json.dumps({"role": role, "content": content})
        self.redis.rpush(f"chat:{session_id}", message)
        
    def get_history(self, session_id: str):
        """전체 히스토리 조회"""
        messages = self.redis.lrange(f"chat:{session_id}", 0, -1)
        return [json.loads(m) for m in messages]

    def delete_session(self, session_id: str):
        self.redis.delete(f"chat:{session_id}")

chat_memory = ChatMemory()

import httpx
from datetime import datetime, timedelta
from jose import jwt, JWTError
from sqlalchemy import select
from sqlalchemy.ext.asyncio import AsyncSession
from models.user import User
from config import get_settings

settings = get_settings()

async def get_google_user_info(code: str):
    """Google 인증 코드로 사용자 정보 조회"""
    async with httpx.AsyncClient() as client:
        # 1. 토큰 교환
        token_res = await client.post("https://oauth2.googleapis.com/token", data={
            "code": code,
            "client_id": settings.GOOGLE_CLIENT_ID,
            "client_secret": settings.GOOGLE_CLIENT_SECRET,
            "redirect_uri": settings.GOOGLE_REDIRECT_URI,
            "grant_type": "authorization_code"
        })
        token_data = token_res.json()
        access_token = token_data.get("access_token")
        
        # 2. 사용자 정보 조회
        user_res = await client.get("https://www.googleapis.com/oauth2/v2/userinfo", headers={
            "Authorization": f"Bearer {access_token}"
        })
        return user_res.json()

def create_access_token(data: dict):
    """JWT 토큰 생성"""
    to_encode = data.copy()
    expire = datetime.utcnow() + timedelta(hours=settings.JWT_EXPIRE_HOURS)
    to_encode.update({"exp": expire})
    return jwt.encode(to_encode, settings.JWT_SECRET_KEY, algorithm=settings.JWT_ALGORITHM)

async def get_or_create_user(db: AsyncSession, google_info: dict) -> User:
    """사용자 조회 또는 생성"""
    result = await db.execute(select(User).where(User.google_id == google_info['id']))
    user = result.scalar_one_or_none()
    
    if not user:
        # 신규 사용자는 여기서 바로 생성하지 않고, 초대코드 확인 로직에서 처리할 수도 있음
        # 본 가이드에서는 자동 가입 처리
        user = User(
            email=google_info['email'],
            name=google_info['name'],
            google_id=google_info['id'],
            picture_url=google_info.get('picture', '')
        )
        db.add(user)
        await db.commit()
        await db.refresh(user)
        
    return user

@router.get("/auth/google")

    """구글 로그인 URL 반환"""
    scope = "https://www.googleapis.com/auth/userinfo.email https://www.googleapis.com/auth/userinfo.profile"
    return {
        "url": f"https://accounts.google.com/o/oauth2/v2/auth?response_type=code&client_id={settings.GOOGLE_CLIENT_ID}&redirect_uri={settings.GOOGLE_REDIRECT_URI}&scope={scope}&access_type=offline"

@router.get("/auth/callback")
async def auth_callback(code: str, db: AsyncSession = Depends(get_db)):
    """구글 콜백 처리"""
    try:
        user_info = await get_google_user_info(code)
        user = await get_or_create_user(db, user_info)
        
        token = create_access_token({"sub": user.email, "user_id": user.id})

backend/api/routes/chat.py
from fastapi import APIRouter, HTTPException
from pydantic import BaseModel
from core.rag_chain import get_rag_chain
from core.memory import chat_memory
import uuid

router = APIRouter()

class ChatRequest(BaseModel):
    message: str
    session_id: str = None

@router.post("/api/chat")
async def chat(request: ChatRequest):
    session_id = request.session_id or str(uuid.uuid4())
    
    # 1. RAG 체인 실행
    chain = get_rag_chain()
    response = await chain.ainvoke({"input": request.message})
    
    answer = response["answer"]
    sources = [doc.metadata.get("source", "Unknown") for doc in response.get("context", [])]
    
    # 2. 메모리 저장
    chat_memory.add_message(session_id, "user", request.message)
    chat_memory.add_message(session_id, "assistant", answer)
    
    return {
        "session_id": session_id,
        "response": answer,
        "sources": list(set(sources)) # 중복 제거
    }

@router.get("/api/chats/{session_id}/history")
def get_history(session_id: str):
    return chat_memory.get_history(session_id)

import streamlit as st
import requests

st.set_page_config(page_title="사내 AI 챗봇", layout="wide")

if "session_id" not in st.session_state:

if "messages" not in st.session_state:

def login():
    st.title("🔐 로그인")
    if st.button("Google 계정으로 로그인"):
        # 실제로는 백엔드에서 받은 URL로 리다이렉트 필요
        # Streamlit 특성상 팝업이나 링크 제공 방식으로 구현
        res = requests.get(f"{API_URL}/auth/google")
        if res.status_code == 200:
            url = res.json()["url"]
            st.markdown(f"[로그인 링크 클릭]({url})", unsafe_allow_html=True)
            st.info("로그인 후 발급받은 토큰을 아래 입력해주세요. (실제 배포시 자동화 가능)")

    token_input = st.text_input("액세스 토큰 입력 (임시 방식)")
    if st.button("토큰 확인"):
        st.session_state.token = token_input
        st.experimental_rerun()

def chat_interface():
    st.title("🤖 Notion RAG Chatbot")
    
    # 사이드바
    with st.sidebar:
        st.header("설정")
        if st.button("새 채팅 시작"):
            st.session_state.session_id = str(uuid.uuid4())
            st.session_state.messages = []
            st.experimental_rerun()
            
    # 채팅 기록 표시
    for msg in st.session_state.messages:
        with st.chat_message(msg["role"]):
            st.markdown(msg["content"])
            
    # 입력 처리
    if prompt := st.chat_input("질문을 입력하세요..."):
        # 사용자 메시지 표시
        with st.chat_message("user"):
            st.markdown(prompt)
        st.session_state.messages.append({"role": "user", "content": prompt})
        
        # API 호출
        with st.spinner("답변 생성 중..."):
            try:
                res = requests.post(
                    f"{API_URL}/api/chat",
                    json={"message": prompt, "session_id": st.session_state.session_id},
                    timeout=60
                )
                if res.status_code == 200:
                    data = res.json()
                    answer = data["response"]
                    sources = data.get("sources", [])
                    
                    full_response = answer
                    if sources:
                        full_response += "\n\n**참고 출처:**\n" + "\n".join([f"- {s}" for s in sources])
                    
                    with st.chat_message("assistant"):
                        st.markdown(full_response)
                    st.session_state.messages.append({"role": "assistant", "content": full_response})
                else:
                    st.error(f"오류 발생: {res.text}")
            except Exception as e:
                st.error(f"연결 오류: {e}")

if not st.session_state.token:
    # 데모를 위해 로그인 패스 가능하도록 수정하거나 로그인 강제
    # 여기서는 로그인 화면 표시
    # login() 
    # 편의상 바로 채팅 화면 (실제 사용시 위 login 주석 해제)
    chat_interface()
else:
    chat_interface()
